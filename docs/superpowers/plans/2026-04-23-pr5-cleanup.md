```python
"""Cancellable WS disconnect watcher (shared by adapter + handler IPC).

Polls ``client.connected`` every ``poll_interval`` seconds. When the
flag flips False (e.g. aiohttp's read loop exits because the server
closed the socket), raises :class:`ConnectionError` so the enclosing
TaskGroup unwinds and the reconnect loop can attempt a fresh connect.
The wall-clock ceiling on disconnect detection is ~``poll_interval``.

Fake test clients without a ``connected`` attribute are tolerated by
treating ``getattr`` misses as "still connected".
"""
from __future__ import annotations

import asyncio
from typing import Any


async def watch_disconnect(client: Any, poll_interval: float = 0.1) -> None:
    while True:
        if not getattr(client, "connected", True):
            raise ConnectionError("ws disconnected")
        await asyncio.sleep(poll_interval)
```

```elixir
  @tag :perf
  test "SessionRouter dispatch latency: 1000 iterations, record p50 / p99" do
    session_id = "perf-p5-10-#{System.unique_integer([:positive])}"
    chat_id = "oc_perf_#{System.unique_integer([:positive])}"
    thread_id = "om_perf_#{System.unique_integer([:positive])}"

    # Bootstrap a real Session subtree + register a test-owned
    # tmux_process pid that simply relays inbound messages back.
    me = self()
    stub_tmux = spawn_link(fn -> relay_loop(me) end)

    {:ok, _session_sup} =
      Esr.SessionsSupervisor.start_session(%{
        session_id: session_id,
        agent_name: "cc",
        dir: "/tmp",
        chat_thread_key: %{chat_id: chat_id, thread_id: thread_id},
        metadata: %{principal_id: "ou_perf"}
      })

    :ok =
      Esr.SessionRegistry.register_session(
        session_id,
        %{chat_id: chat_id, thread_id: thread_id},
        %{feishu_chat_proxy: me, tmux_process: stub_tmux}
      )

    env = %{
      "payload" => %{
        "chat_id" => chat_id,
        "thread_id" => thread_id,
        "text" => "perf probe"
      }
    }

    n = 1000
    samples =
      for _ <- 1..n do
        t0 = System.monotonic_time(:microsecond)
        send(Esr.SessionRouter, {:inbound_event, env})
        # Wait for the stub to observe the message (one hop).
        assert_receive {:relay, _msg}, 500
        System.monotonic_time(:microsecond) - t0
      end

    sorted = Enum.sort(samples)
    p50 = Enum.at(sorted, div(n, 2))
    p99 = Enum.at(sorted, div(n * 99, 100))

    IO.puts(
      "perf: SessionRouter dispatch latency (n=#{n}) p50=#{p50}µs p99=#{p99}µs"
    )

    # Only regression-guard the p99; p50 is purely informational.
    # Threshold is generous because this is a synthetic stub — the
    # real goal is to capture a stable number for PR-6 to compare
    # against.
    assert p99 < 10_000, "p99 dispatch latency over 10ms is suspicious"

    # Persist the numbers for PR-6 to pick up without re-running the
    # smoke; a plain tmpfile is sufficient.
    File.write!(
      System.tmp_dir!() <> "/esr-pr5-perf-baseline.tsv",
      "p50_us\tp99_us\n#{p50}\t#{p99}\n"
    )

    Esr.SessionRegistry.unregister_session(session_id)
  end

  defp relay_loop(dest) do
    receive do
      msg ->
        send(dest, {:relay, msg})
        relay_loop(dest)
    end
  end
```
